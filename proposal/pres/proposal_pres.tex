\section{Useful Tools}
\frame{\frametitle{Shannon's Information Theory}
\begin{align*}
H(X)    & = \int_{\mathcal{X}} p(x) \ln p(x) \, dx
        & \mbox{(Entropy)}  \\
D(X\|Y) & = \int_{\mathcal{X}}
                \int_{\mathcal{Y}} p(x) \ln \frac{p(x)}{p(y)} \, dy \, dx
        & \mbox{(KL-Divergence)}  \\
I(X;Y)  & = D(p(x,y)\|p(x)p(y))
        & \mbox{(Mutual Information)}  \\
H(X|Y)  & = H(X) - I(X;Y)
        & \mbox{(Conditional Entropy)}
\end{align*}
}
